Panda Secret has been deployed!

1. Get the application URL:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "panda-secret.fullname" . }} 8080:{{ .Values.service.port }}
  Then open http://localhost:8080
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc {{ include "panda-secret.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if eq .Values.service.type "LoadBalancer" }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get svc {{ include "panda-secret.fullname" . }} -w
{{- end }}

2. Check deployment status:
  kubectl get pods -l app.kubernetes.io/name={{ include "panda-secret.name" . }}

3. View HPA status:
{{- if .Values.autoscaling.enabled }}
  kubectl get hpa {{ include "panda-secret.fullname" . }}
{{- else }}
  Autoscaling is disabled. Enable it by setting autoscaling.enabled=true
{{- end }}
